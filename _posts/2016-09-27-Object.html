Object Oriented Programming

Make a simple car drive off from the right end of the screen.

{% highlight js %} color c; float xpos; float ypos; float xspeed; void setup() { rectMode(CENTER); size(480, 270); c = color(175); xpos = width/2; ypos = height/2; xspeed = 1; } void draw() { background(255); stroke(0); fill(c); rect(xpos, ypos, 20, 10); xpos = xpos + xspeed; if (xpos > width) { xpos = 0; } } {% endhighlight%}

Now, make two more cars.

{% highlight js %} color c; float xpos; float ypos; float xspeed; color c1; float xpos1; float ypos1; float xspeed1; color c2; float xpos2; float ypos2; float xspeed2; void setup() { rectMode(CENTER); size(480, 270); c = color(175); xpos = 0; ypos = height/2+30; xspeed = 1; c1 = color(200); xpos1 = 0; ypos1 = height/2; xspeed1 = 10; c2 = color(230); xpos2 = 0; ypos2 = height/2-15; xspeed2 = 7; } void draw() { background(255); stroke(0); fill(c); rect(xpos, ypos, 20, 10); fill(c1); rect(xpos1, ypos1, 30, 8); fill(c2); rect(xpos2, ypos2, 15, 12); xpos = xpos + xspeed; xpos1 = xpos1 + xspeed1; xpos2 = xpos2 + xspeed2; if (xpos > width) { xpos = 0; } if (xpos1 > width) { xpos1 = 0; } if (xpos2 > width) { xpos2 = 0; } } {% endhighlight %}
